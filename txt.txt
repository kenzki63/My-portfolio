# /scripts/script.js
```javascript
// Upgraded JS: GSAP powered reveals, cursor interactions, hover 3D, parallax, selection pulse
(() => {
  // GSAP setup
  gsap.registerPlugin(ScrollTrigger);

  // Basic DOM refs
  const cursor = document.getElementById('cursor');
  const ripple = document.getElementById('cursorRipple');
  const hoverTargets = document.querySelectorAll('.hover-target, .nav-link, .btn, .project-card, .skill-group, .project-media img, .contact-card');

  // initial mouse position
  let mouse = {x: window.innerWidth/2, y: window.innerHeight/2};
  let pos = {x: mouse.x, y: mouse.y};

  // Smooth cursor motion using GSAP ticker
  function moveCursor(){
    pos.x += (mouse.x - pos.x) * 0.18;
    pos.y += (mouse.y - pos.y) * 0.18;
    gsap.set(cursor, {x: pos.x, y: pos.y});
    gsap.set(ripple, {x: mouse.x, y: mouse.y});
  }
  gsap.ticker.add(moveCursor);

  window.addEventListener('mousemove', (e)=>{
    mouse.x = e.clientX; mouse.y = e.clientY;
  });

  // click: shrink briefly
  window.addEventListener('mousedown', ()=>{
    gsap.to(cursor, {scale:0.5, duration:0.12, ease:'power3.out'});
    // ripple
    gsap.fromTo(ripple, {opacity:0.8, scale:0.3}, {opacity:0, scale:1.6, duration:0.7, ease:'power2.out'});
  });
  window.addEventListener('mouseup', ()=>{
    gsap.to(cursor, {scale:1, duration:0.28, ease:'elastic.out(1,0.6)'});
  });

  // Hover interactions: float-up 3D effect using GSAP on enter/leave
  hoverTargets.forEach(el=>{
    el.addEventListener('mouseenter', ()=>{
      gsap.to(el, {y:-8, z:10, duration:0.38, ease:'power3.out'});
      gsap.to(cursor, {scale:1.4, duration:0.18});
    });
    el.addEventListener('mouseleave', ()=>{
      gsap.to(el, {y:0, z:0, duration:0.45, ease:'power3.out'});
      gsap.to(cursor, {scale:1, duration:0.25});
    });
  });

  // Reveal animations for all elements with .reveal
  gsap.utils.toArray('.reveal').forEach(el=>{
    gsap.fromTo(el, {y:18, autoAlpha:0}, {
      y:0, autoAlpha:1, duration:0.9, ease:'power2.out',
      scrollTrigger:{trigger:el, start:'top 85%', toggleActions:'play none none none'}
    });
  });

  // Staggered reveal for project cards
  ScrollTrigger.batch('.projects-grid .project-card', {
    interval: 0.12,
    batchMax: 6,
    onEnter: batch => gsap.to(batch, {y: 0, autoAlpha: 1, stagger: 0.08, duration: 0.8, ease: 'power3.out'}),
    onLeave: batch => gsap.to(batch, {autoAlpha:0}),
    start: 'top 85%'
  });

  // Parallax: elements with data-parallax
  const parallaxEls = document.querySelectorAll('[data-parallax]');
  window.addEventListener('scroll', ()=>{
    parallaxEls.forEach(el=>{
      const factor = parseFloat(el.dataset.parallax) || 0.02;
      const offset = window.scrollY * factor;
      gsap.to(el, {y: offset, duration: 0.6, ease: 'power1.out'});
    });
  });

  // Background blob subtle follow on mouse
  const blobs = document.querySelectorAll('.bg-blob');
  window.addEventListener('mousemove', (e)=>{
    const cx = (e.clientX / window.innerWidth - 0.5) * 30;
    const cy = (e.clientY / window.innerHeight - 0.5) * 30;
    blobs.forEach((b, i)=>{
      const depth = (i+1) * 6;
      gsap.to(b, {x: cx * (i+0.6), y: cy * (i+0.6), duration: 1.2, ease: 'power2.out'});
    });
  });

  // Hide cursor on inactivity
  let idleTimer;
  function resetIdle(){
    clearTimeout(idleTimer);
    gsap.to(cursor, {autoAlpha:1, duration:0.4});
    idleTimer = setTimeout(()=>gsap.to(cursor, {autoAlpha:0, duration:0.9}), 3000);
  }
  window.addEventListener('mousemove', resetIdle);
  resetIdle();

  // Text selection pulse: listen to selectionchange
  document.addEventListener('selectionchange', ()=>{
    const sel = document.getSelection().toString();
    if(sel && sel.length>0){
      // pulse cursor
      gsap.fromTo(cursor, {scale:1}, {scale:1.9, duration:0.28, yoyo:true, repeat:1, ease:'power2.out'});
    }
  });

  // Prevent native pointer showing on interactive elements: add style where required
  document.querySelectorAll('a, button, input, textarea, label').forEach(el=>{
    el.style.cursor = 'none';
  });

  // Smooth nav scroll
  document.querySelectorAll('a[href^="#"]').forEach(a=>{
    a.addEventListener('click', (e)=>{
      e.preventDefault();
      const t = document.querySelector(a.getAttribute('href'));
      if(!t) return;
      t.scrollIntoView({behavior:'smooth', block:'start'});
    });
  });

  // Footer year
  document.getElementById('year').textContent = new Date().getFullYear();

  // Accessibility: ensure focus outlines visible for keyboard users
  document.addEventListener('keydown', (e)=>{
    if(e.key === 'Tab'){
      document.documentElement.classList.add('show-focus');
    }
  });

})();
```

# Notes
- This replacement uses **GSAP + ScrollTrigger** (CDN links included in the HTML). No other external libs.
- Cursor hides the default pointer and fully replaces it. Interactive elements set `cursor:none` so native pointer doesn't reappear.
- Hover/float effects are applied via JS for better control and physics-like motion with GSAP.
- Text selection pulse triggers when a user highlights text.
- If you want keyboard accessibility improvements (focus styles that work with the custom cursor), I can add explicit focus-visible styles next.
- If you want the project packed as a ZIP, I can generate that next.
